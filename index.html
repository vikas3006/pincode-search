          if(fly[key]){fly[key].then(all=>{const kws=q.split(',').map(k=>k.trim().toLowerCase()).filter(Boolean);const r=isPin?all:filt(all,kws.join(' '));const merged=merge(r,loc);if(merged.length)updateURLFromResults(merged,q);res.innerHTML=`<div class="pcs-status">Found <strong>${merged.length}</strong> result${merged.length!==1?'s':''} for "<strong>${esc(q)}</strong>"</div>`;renderRows(merged);if(merged.length)findNearby(merged[0]);}).catch(()=>{});}

        }else{res.innerHTML="<p style='color:#e74c3c;margin-top:10px'>Connection is slow — please try again.</p>";}

      }else if(e.name!=='AbortError'){

        if(loc.length){res.innerHTML=`<div class="pcs-status">Showing ${loc.length} local result${loc.length>1?'s':''} (API unavailable)</div>`;renderRows(loc);}

        else res.innerHTML="<p style='color:#e74c3c;margin-top:10px'>Error fetching data. Please try again.</p>";

      }

    });

}


/* NEW: smarter URL once we know state/district */

function updateURLFromResults(results,q){

  const isPin=/^\d{6}$/.test(q);

  const first=results[0];

  if(!first)return;

  if(isPin){

    /* pincode search → full hierarchy URL */

    setURL(toSlug(first.State)+'/'+toSlug(first.District)+'/'+toSlug(first.Name)+'/'+first.Pincode);

  } else if(results.every(r=>r.District===first.District)){

    /* all same district → district-level URL */

    setURL(toSlug(first.State)+'/'+toSlug(first.District));

  } else if(results.every(r=>r.State===first.State)){

    /* all same state → state-level URL */

    setURL(toSlug(first.State));

  }

  /* mixed states → keep slug of query */

}
